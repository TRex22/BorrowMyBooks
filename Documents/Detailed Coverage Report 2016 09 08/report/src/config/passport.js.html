            usernameField: 'email',
            passReqToCallback: true

        function(req, email, password, done) {
            process.nextTick(function() {
                User.findOne({ 'user.email': email }, function(err, user) {
                    if (err) {
                        return done(err);

                        return done(null, false, req.flash('error', 'User does not exist.'));
